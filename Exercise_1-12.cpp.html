#include <stdio.h>

#define IN 1
#define OUT 0

// count lines, words, and characters in input

int main()
{
	int state, c, spaceCounter;
	state = IN;
	spaceCounter = 0;

	while ((c = getchar()) != EOF)
	{
		if ( c ==  ' ' || c == '\n' || c == '\t')
		{
			state = OUT;
			++spaceCounter;
		}
		if (state == IN)
		{
			putchar(c);
		}
		else if (state == OUT)
		{
			state = IN;
			if (spaceCounter <= 1)
			{
				printf("\n");
			}
			spaceCounter = 0;
		}
	}
}
